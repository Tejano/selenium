Sub ImportObjectsToACCDB()
    Dim objTypeArray As Variant
    Dim i As Integer
    Dim importPath As String
    Dim fileName As String

    importPath = "C:\ExportedObjects\" ' Ensure this matches the export path

    ' Array of object types to import
    objTypeArray = Array(acForm, acReport, acMacro, acModule)

    For i = LBound(objTypeArray) To UBound(objTypeArray)
        Select Case objTypeArray(i)
            Case acForm
                fileName = Dir(importPath & "*.txt")
                While fileName <> ""
                    DoCmd.LoadFromText acForm, Left(fileName, Len(fileName) - 4), importPath & fileName
                    fileName = Dir
                Wend
            Case acReport
                fileName = Dir(importPath & "*.txt")
                While fileName <> ""
                    DoCmd.LoadFromText acReport, Left(fileName, Len(fileName) - 4), importPath & fileName
                    fileName = Dir
                Wend
            Case acMacro
                fileName = Dir(importPath & "*.txt")
                While fileName <> ""
                    DoCmd.LoadFromText acMacro, Left(fileName, Len(fileName) - 4), importPath & fileName
                    fileName = Dir
                Wend
            Case acModule
                fileName = Dir(importPath & "*.txt")
                While fileName <> ""
                    DoCmd.LoadFromText acModule, Left(fileName, Len(fileName) - 4), importPath & fileName
                    fileName = Dir
                Wend
        End Select
    Next i

    MsgBox "Import completed!"
End Sub

Sub LinkSQLServerTables()
    Dim db As DAO.Database
    Dim tdf As DAO.TableDef
    Dim serverName As String
    Dim databaseName As String
    Dim userId As String
    Dim password As String
    Dim connString As String

    serverName = "YourServerName"
    databaseName = "YourDatabaseName"
    userId = "YourUserId"
    password = "YourPassword"

    connString = "ODBC;Driver={SQL Server};Server=" & serverName & ";Database=" & databaseName & ";UID=" & userId & ";PWD=" & password & ";"

    Set db = CurrentDb()

    ' Example of linking a table
    ' You will need to iterate through all tables you want to link
    Set tdf = db.CreateTableDef("YourTableName")
    tdf.Connect = connString
    tdf.SourceTableName = "YourTableName"
    db.TableDefs.Append tdf

    MsgBox "Tables linked successfully!"
End Sub
